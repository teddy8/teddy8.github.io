## 소스코드
``` cpp
// 3개의 정수 입력받아 작은 수 출력하기
// 단 조건문을 사용하지 않고 삼항 연산자 사용
// input : 1 2 3
// output : 1
#include <iostream>

using namespace std;

int main()
{
	int n1, n2, n3;
	cin >> n1 >> n2 >> n3;
	cout << (n1 < n2 && n1 < n3 ? n1 : (n2 < n3) ? n2 : n3);
	return 0;
}
```
